    class Node {
      constructor(x, y, xdirection, ydirection) {
        this.wall = false;
        this.x = x;
        this.y = y;
        this.xdirection = xdirection;
        this.ydirection = ydirection;
        this.r = floor(random(100, 255));
        this.g = floor(random(100, 255));
        this.b = floor(random(100, 255));
      }
    
      displayNode() {
        fill(this.r, this.g, this.b);
        //rect(this.x, this.y, scl, scl);
        textSize(16);
        fill(255);
        textSize(25);
        text("ðŸ˜Š", this.x + 2, this.y + 15);
      }
      checkBounds() {
        this.x = constrain(this.x, 0, width - scl);
        this.y = constrain(this.y, 0, height - scl);
      }
      updatePosition() {
        this.x = this.x + this.xdirection * scl;
        this.y = this.y + this.ydirection * scl;
        this.checkBounds();
      }
      updateDirection() {
        if (this.prev != null) {
          this.xdirection = this.prev.xdirection;
          this.ydirection = this.prev.ydirection;
        }
      }
    }
    class LinkedList {
      constructor(theList) {
        this.head = theList;
        this.end = theList;
        this.size = 0;
      }
    
      updateDirections() {
        var current = this.head;
        current = this.end;
        while (current != null) {
          current.updateDirection();
          current = current.prev;
        }
      }
      display() {
        var current = new Node();
        current = this.head;
        while (current != null) {
          current.displayNode();
          current.updatePosition();
          current = current.next;
        }
      }
      dir(x, y) {
        this.head.xdirection = x;
        this.head.ydirection = y;
      }
    
      addNodeInTheHead() {
        var node = new Node();
        node = this.head;
        if (this.head.ydirection == -1) {
          node = new Node(
            this.head.x,
            this.head.y - scl,
            this.head.xdirection,
            this.head.ydirection
          );
        }
        if (this.head.xdirection == 1) {
          node = new Node(
            this.head.x + scl,
            this.head.y,
            this.head.xdirection,
            this.head.ydirection
          );
        }
        if (this.head.ydirection == 1) {
          node = new Node(
            this.head.x,
            this.head.y + scl,
            this.head.xdirection,
            this.head.ydirection
          );
        }
        if (this.head.xdirection == -1) {
          node = new Node(
            this.head.x - scl,
            this.head.y,
            this.head.xdirection,
            this.head.ydirection
          );
        }
        node.next = this.head;
        this.head.prev = node;
        this.head = node;
      }
      move() {
        this.addNodeInTheHead();
        this.deleteNodeInTail();
      }
      addFront(node) {
        node.next = this.head;
        if (this.size == 0) {
          this.end = node;
        } else {
          this.head.prev = node;
        }
        this.head = node;
        this.size++;
      }
    
      deleteNodeInTail() {
        //this.end = null;
        //node.next = null;
        this.end = node;
        var node = this.end.prev;

        foodx = foodPosx * scl;

class NodeList {
  constructor(theList) {
    this.theList = theList;
    this.x = 45;
    this.y = 320;
    this.nl;
    //for the lines below
    this.x1 = this.x; //where it starts
    this.y1 = this.y;

    this.x2 = this.x; //where the line is going
    this.y2 = this.y;

    this.x3 = this.x; //where line is going but draws/subtracts distance from x1
    //x3-x1
    this.y3 = this.y;

    this.countNode = 0;
  }

  add() {
    this.countNode += 1;
    var newNode = new Node(this.theList, this.x, this.y, this.countNode);
    this.x += 45;
    this.x3 += 45;
    newNode.setNext(this.theList);
    this.theList = newNode;
    if (this.countNode > 7) {
      this.countNode = 0;
      newNode = null;
      nl = new NodeList(newNode);
    }
  }

  count() {
    var count = 0;
    var nodes = this.theList;
    while (nodes != null) {
      nodes = nodes.next;
      count++;
    }
    return count;
  }

  displayList() {
    var nodes = this.theList;
    while (nodes != null) {
      nodes.display();
      nodes = nodes.next;
    }
  }
  displayLine() {
    this.x1 += (this.x3 - this.x1) * 0.06;
    this.y1 += (this.y3 - this.y1) * 0.06;
    stroke("#FF581B");
    strokeWeight(2);
    line(this.x1, this.y1, this.x2, this.y2);
  }
}

class Taxi {
  constructor(newID, tempX, tempY, tempPaint, tempSpeed) {
    this.taxiID = newID;
    this.head = null;
    this.next = null;
    this.x = tempX;

  }
}
class TaxiRank {
  constructor(newNode) {
    var head = new Taxi();
    this.head = null;
    var tail = new Taxi();
    tail = null;
  }

  add(newNode) {
    if (this.head == null) {
      this.head = newNode;
      //this.tail = newNode;
      return;
    } else {
      if (newNode != null) {
        var current = new Taxi();
        current = this.head;
        while (current.next != null) {
          current = current.next;
        }
        current.next = newNode;
        //tail=newNode;
        newNode.prev = current;
      }
    }
  }

  list() {
    var countY = height / 2;
    var current = new Taxi();
    current = this.head;
    while (current != null) {
      current.displayCar();
      fill(255);
      textSize(14);
      text(current.taxiID, current.x - 18, current.y - 10);
      current = current.next;
      countY += 20;
    }
  }

  get() {
    if (this.head == null) {
      return null; //no cabs
    } else {
      var returnVal = new Taxi();
      returnVal = this.head;
      this.head = this.head.next; //move cabs along one line
      //this.head.prev = null;
      return returnVal;
    }
  }

  isEmpty() {
    if (head == null) return true;
    return false;
  }
}
